namespace = roman_question

# Rome not held by the Pope
roman_question.1 = {
	type = country_event
	placement = scope:roman_question_state
	title = roman_question.1.t
	desc = roman_question.1.d
	flavor = roman_question.1.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/southamerica_christianity.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/engineer_blueprint"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	trigger = {
		NOT = { has_global_variable = roman_question_var }
		any_scope_state = {
			state_region = s:STATE_LAZIO
		}
		NOT = {
			AND = {
				has_law = law_type:law_theocracy
				OR = {
					AND = {
						exists = c:PAP
						c:PAP = ROOT
					}
					AND = {
						exists = c:ITA
						c:ITA = ROOT
					}
				}
			}
		}
	}

	immediate = {
		set_global_variable = roman_question_var
		set_variable = roman_question_country_var
		random_scope_state = {
			limit = {
				state_region = s:STATE_LAZIO
			}
			save_scope_as = roman_question_state
		}
		if = {
			limit = {
				AND = {
					exists = c:PAP
					c:PAP = ROOT
				}
			}
			ig:ig_devout = {
				remove_ideology = ideology_papal_moralist
				add_ideology = ideology_moralist
			}
			ig:ig_landowners = {
				remove_ideology = ideology_papal_paternalistic
			}
			if = {
				limit = {
					has_law = law_type:law_monarchy
				}
				ig:ig_landowners = {
					add_ideology = ideology_paternalistic
				}
			}
			else = {
				ig:ig_landowners = {
					add_ideology = ideology_republican_paternalistic
				}
			}
			ig:ig_intelligentsia = {
				remove_ideology = ideology_papal_anti_clerical
				add_ideology = ideology_anti_clerical
			}
		}
	}

	option = {
		# give the Pope just the Vatican and pay him to settle things
		name = roman_question.1.a
		default_option = yes

		trigger = {
			religion = rel:catholic
		}

		add_modifier = {
			name = papal_treaty_cost_modifier
			months = short_modifier_time
			multiplier = money_amount_multiplier_small
		}
		ig:ig_devout = {
			add_modifier = {
				name = papal_treaty_mod
				months = normal_modifier_time
			}
		}
	}
	option = {
		name = roman_question.1.b1
		# kick the Pope out. get lost, nerd

		trigger = {
			religion = rel:catholic
		}
		custom_tooltip = {
			text = roman_question.1.tt
			set_global_variable = pope_exiled_var
		}
		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_theocracy
					value > neutral
				}
			}
			add_modifier = {
				name = pope_exiled_mod
				months = short_modifier_time
			}
		}
	}
	option = {
		name = roman_question.1.b2
		default_option = yes
		# not Catholic

		trigger = {
			NOT = {
				religion = rel:catholic
			}
		}
		custom_tooltip = {
			text = roman_question.1.tt
			set_global_variable = pope_exiled_var
		}
		ig:ig_devout = {
			add_modifier = {
				name = pope_good_riddance_mod
				months = normal_modifier_time
			}
		}
	}
	option = {
		name = roman_question.1.c1
		# give Rome back to the Pope

		trigger = {
			religion = rel:catholic
			exists = c:PAP
			NOT = {
				c:PAP = ROOT
			}
			c:PAP = {
				has_law = law_type:law_theocracy
			}
		}

		scope:roman_question_state = {
			set_state_owner = c:PAP
		}
		c:PAP = {
			change_relations = {
				country = ROOT
				value = 30
			}
		}

		ig:ig_devout = {
			add_modifier = {
				name = papal_treaty_mod
				months = normal_modifier_time
			}
		}
	}
	option = {
		name = roman_question.1.c2
		# recreate the Papal States

		trigger = {
			religion = rel:catholic
			NOT = {
				exists = c:PAP
			}
		}
		custom_tooltip = {
			text = recreate_papal_states
			create_country = {
				tag = PAP
				origin = this
				state = scope:roman_question_state
				on_created = {
					if = {
						limit = {
							NOT = {
								has_ruling_interest_group = ig_devout
							}
						}
						ig:ig_devout = {
							add_ruling_interest_group = yes
						}
					}
					if = {
						limit = {
							NOT = {
								has_law = law_type:law_theocracy
							}
						}
						activate_law = law_type:law_theocracy
					}
					if = {
						limit = {
							NOT = {
								has_law = law_type:law_state_religion
							}
						}
						activate_law = law_type:law_state_religion
					}
					if = {
						limit = {
							NOT = {
								has_law = law_type:law_oligarchy
							}
						}
						activate_law = law_type:law_oligarchy
					}
					trigger_event = { id = roman_question.2 days = 0 popup = yes }
					change_relations = {
						country = ROOT
						value = 30
					}
				}
			}
			ROOT = {
				create_diplomatic_pact = {
					country = c:PAP
					type = protectorate
				}
			}
		}

		ig:ig_devout = {
			add_modifier = {
				name = papal_treaty_mod
				months = normal_modifier_time
			}
		}
	}
}

# set up the new Papal States
roman_question.2 = {
	type = country_event
	placement = ROOT
	title = roman_question.2.t
	desc = roman_question.2.d
	flavor = roman_question.2.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/southamerica_christianity.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	option = {
		name = roman_question.2.a
		default_option = yes
		hidden_effect = {
			ruler = {
				kill_character = yes
			}
			if = {
				limit = {
					any_scope_character = {
						is_heir = yes
					}
				}
				heir = {
					kill_character = yes
				}
			}
		}
		create_character = {
			culture = cu:south_italian
			ruler = yes
			birth_date = 1810.3.2
			interest_group = ig_devout
			traits = {
			}
		}
		ig:ig_landowners = {
			set_interest_group_name = ig_roman_landowners
			remove_ideology = ideology_paternalistic
			add_ideology = ideology_papal_paternalistic
		}
		ig:ig_intelligentsia = {
			remove_ideology = ideology_anti_clerical
			add_ideology = ideology_papal_anti_clerical
		}
		ig:ig_devout = {
			set_ig_bolstering = yes
			set_interest_group_name = ig_roman_curia
			remove_ideology = ideology_moralist
			add_ideology = ideology_papal_moralist
		}
	}
}

roman_question.3 = {
	type = country_event
	placement = scope:papal_donation_state
	title = roman_question.3.t
	desc = roman_question.3.d
	flavor = roman_question.3.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/southamerica_christianity.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	trigger = {
		has_global_variable = pope_exiled_var
		NOT = {
			has_variable = roman_question_country_var
		}
		NOT = {
			exists = c:PAP
		}
		religion = rel:catholic
		ig:ig_devout = {
			is_marginal = no
		}
		any_scope_state = {
			is_incorporated = yes
			is_capital = no
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 5
			}
			any_scope_pop = {
				religion = ROOT.religion
				is_pop_type = clergymen
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 5
				}
				any_scope_pop = {
					religion = ROOT.religion
					is_pop_type = clergymen
				}
				is_capital = no
			}
			save_scope_as = papal_donation_state
		}
		ig:ig_devout = {
			save_scope_as = church_scope
		}
		set_variable = roman_question_country_var
	}

	cancellation_trigger = {
		exists = c:PAP
	}

	option = {
		# nope
		name = roman_question.3.a
		default_option = yes

		ig:ig_devout = {
			add_modifier = {
				name = pope_refused_mod
				months = normal_modifier_time
			}
		}
	}

	option = {
		# recreate the Papal States
		name = roman_question.3.b

		custom_tooltip = {
			text = recreate_papal_states
			create_country = {
				tag = PAP
				origin = this
				state = scope:papal_donation_state
				on_created = {
					if = {
						limit = {
							NOT = {
								has_ruling_interest_group = ig_devout
							}
						}
						ig:ig_devout = {
							add_ruling_interest_group = yes
						}
					}
					if = {
						limit = {
							NOT = {
								has_law = law_type:law_theocracy
							}
						}
						activate_law = law_type:law_theocracy
					}
					if = {
						limit = {
							NOT = {
								has_law = law_type:law_state_religion
							}
						}
						activate_law = law_type:law_state_religion
					}
					if = {
						limit = {
							NOT = {
								has_law = law_type:law_oligarchy
							}
						}
						activate_law = law_type:law_oligarchy
					}
					trigger_event = { id = roman_question.2 days = 0 popup = yes }
					change_relations = {
						country = ROOT
						value = 30
					}
				}
			}
			ROOT = {
				create_diplomatic_pact = {
					country = c:PAP
					type = protectorate
				}
			}
		}

		ig:ig_devout = {
			add_modifier = {
				name = papal_treaty_mod
				months = normal_modifier_time
			}
		}
		remove_global_variable = pope_exiled_var
	}
