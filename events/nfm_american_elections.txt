#1836
nfm_american_elections.1 = {
	#1836 Nomination 
	type = country_event
	placement = ROOT
	title = nfm_american_elections.1.t
	desc = nfm_american_elections.1.d
	flavor = nfm_american_elections.1.f
	event_image = {
		video = "gfx/event_pictures/europenorthamerica_capitalists_meeting.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/capitalists_meeting"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	duration = 3
	is_popup = yes
	trigger = {
		exists = c:USA
		c:USA = ROOT
		in_election_campaign = yes
		NOT = {
			has_variable = 1836_dem_nomination_var
		}
		year = 1836
		NOT = {
			has_variable = end_of_two_party_system_var
		}
	}
	immediate = {
		set_variable = 1836_dem_nomination_var
		create_character = {
			#
			first_name = Martin
			last_name = Van_Buren
			culture = cu:yankee
			religion = protestant
			ideology = ideology_abolitionist			#Free Soil Party
			interest_group = ig_rural_folk
			birth_date = 1782.12.5
			dna = ecchi_dna_martin_van_buren
			traits = {
				cautious
				basic_diplomat
			}
			on_created = {
				set_variable = {
					name = is_martin_van_buren					#give the character a variable
				}
				save_scope_as = character_to_display_martin_van_buren
			}
		}
		random_scope_character = {
			limit = {
				has_variable = is_martin_van_buren
			}
			#use the variable to scope to that specific character
			save_scope_as = character_to_display_martin_van_buren			#save the scope here
		}
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_rural_folk
			}
			leader = {
				save_scope_as = character_to_display_rural_folk_leader
			}
		}
	}
	option = {
		name = nfm_american_elections.1.a
		default_option = yes
		ai_chance = {
			base = 0
		}
		set_variable = 1836_dem_nomination_var_martin_van_buren
		ig:ig_rural_folk = {
			party = {
				add_momentum = 0.2
			}
		}
	}
}
