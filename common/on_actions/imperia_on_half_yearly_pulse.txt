on_imperia_half_yearly_pulse_country = {
	effect = {
		# Resets the global list twice per year only
		# Instead of [every_country -> every_country] x yearly
		if = {
			limit = {
				has_global_highest_gdp = yes
				has_global_variable = labor_movement_researched
			}
			clear_global_variable_list ?= council_republics_list
			every_country = {
				limit = {
					has_law = law_type:law_council_republic
					country_rank > rank_value:minor_power
				}
				add_to_global_variable_list = {
					name = council_republics_list
					target = THIS
				}
			}
		}
		# If your heir is above the age of 2, and is still a child
		if = {
			limit = {
				heir ?= {
					age >= 2
					has_trait = trait_child
					# Dont pop for pre-generated heirs that werent "born"
					OR = {
						has_variable = raised_by_clergy
						has_variable = raised_by_nobles
						has_variable = raised_by_burghers
						has_variable = raised_by_ruler
					}
					NOT = {
						has_variable = stop_education
					}
				}
			}
			# Do a education progress tick
			progress_heir_education = yes
		}
		# Else if the heir has grown up, but still has the variables
		else_if = {
			limit = {
				heir ?= {
					NOT = {
						has_trait = trait_child
					}
					OR = {
						has_variable = raised_by_clergy
						has_variable = raised_by_nobles
						has_variable = raised_by_burghers
						has_variable = raised_by_ruler
					}
				}
			}
			heir = {
				remove_all_education_variables_from_character = yes
			}
		}
		if = {
			limit = {
				ruler = {
					age >= 2
					has_trait = trait_child
					has_variable = raised_by_ruler
				}
				remove_variable = raised_by_ruler
			}
		}
	}
}
