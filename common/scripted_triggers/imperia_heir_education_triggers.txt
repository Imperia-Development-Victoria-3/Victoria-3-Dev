WARDEN_can_teach_WARD_artillery = {
	$WARD$.owner = {
		has_technology_researched = artillery
	}
	OR = {
		# Lvl 1 available
		AND = {
			$WARD$ = {
				NOR = {
					has_modifier = basic_artillery_commander_education
					has_modifier = experienced_artillery_commander_education
					has_modifier = expert_artillery_commander_education
				}
			}
			$WARDEN$ = {
				OR = {
					has_trait = basic_artillery_commander
					has_trait = experienced_artillery_commander
					has_trait = expert_artillery_commander
				}
			}
		}
		# Lvl 2 available
		AND = {
			$WARD$ = {
				has_modifier = basic_artillery_commander_education
			}
			$WARDEN$ = {
				OR = {
					has_trait = experienced_artillery_commander
					has_trait = expert_artillery_commander
				}
			}
		}
		# Lvl 3 available
		AND = {
			$WARD$ = {
				has_modifier = experienced_artillery_commander_education
			}
			$WARDEN$ = {
				has_trait = expert_artillery_commander
			}
		}
	}
}
