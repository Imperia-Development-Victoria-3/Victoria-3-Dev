frankfurt_conference_has_majority_for_republic = {
	OR = {
		# Either theres more republic voters and monarchy voters
		imperia_frankfurt_conference_republic_voters_value > imperia_frankfurt_conference_monarchy_voters_value
		AND = {
			# Or the two are equal
			imperia_frankfurt_conference_republic_voters_value = imperia_frankfurt_conference_monarchy_voters_value
			# But the owner of Frankfurt is voting republican
			p:x11C0A0 = {
				owner = {
					has_variable = frankfurt_conference_voted_republic
				}
			}
		}
	}
}

frankfurt_conference_has_majority_for_oligarchy = {
	# Either...
	OR = {
		# Oligarchy beat the 2 others
		AND = {
			imperia_frankfurt_conference_oligarchy_voters_value > imperia_frankfurt_conference_wealth_voters_value
			imperia_frankfurt_conference_oligarchy_voters_value > imperia_frankfurt_conference_universal_voters_value
		}
		# Or oligarchy didnt lose, and Frankfurt voted for it
		# (means a tie happened with at least one of them, possibly both)
		AND = {
			NOT = {
				imperia_frankfurt_conference_oligarchy_voters_value < imperia_frankfurt_conference_wealth_voters_value
			}
			NOT = {
				imperia_frankfurt_conference_oligarchy_voters_value < imperia_frankfurt_conference_universal_voters_value
			}
			p:x11C0A0 = {
				owner = {
					has_variable = frankfurt_conference_voted_oligarchy
				}
			}
		}
	}
}

frankfurt_conference_prince_will_join_gutter_king = {
	# scope = GUTTER KING
	# root = PRINCE
	OR = {
		country_rank = rank_value:great_power
		AND = {
			country_rank = rank_value:major_power
			root.country_rank <= rank_value:major_power
		}
		AND = {
			country_rank = rank_value:minor_power
			root.country_rank <= rank_value:major_power
		}
		AND = {
			country_rank = rank_value:insignificant_power
			root.country_rank <= rank_value:minor_power
			# Special mention for Bavaria and Hanover, who didnt join the Frankfurt-led German empire...
			trigger_if = {
				limit = { 
					exists = c:BAV
				}
				root != c:BAV
			}
			trigger_if = {
				limit = { 
					exists = c:HAN
				}
				root != c:HAN
			}
		}
	}
}
