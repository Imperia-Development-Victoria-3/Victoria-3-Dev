				has_ideology = ideology:ideology_liberal_leader
			}
		}
		multiply = 3
	}
	if = {
		limit = {
			OR = {
				has_ideology = ideology:ideology_royalist
				has_ideology = ideology:ideology_authoritarian
			}
		}
		divide = 3
	}
}
