							owner = root
							is_incorporated = yes

						shares_heritage_and_other_trait_with_any_primary_culture = root
					}
				}
			}
		}
		multiply = 0
	}
	if = {
		limit = {
			has_law = law_type:law_anti_imperialist

				OR = {
					has_law = law_type:law_no_colonial_affairs

		}
