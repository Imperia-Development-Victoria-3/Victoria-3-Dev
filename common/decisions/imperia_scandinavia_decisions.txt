# An Offer for Guadeloupe
guadeloupe.1 = {
	type = country_event
	placement = ROOT

	title = guadeloupe.1.t
	desc = guadeloupe.1.d
	flavor = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:guadeloupe_seller = {
						c:FRA ?= this
					}
				}
				desc = guadeloupe.1.f_fra
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = guadeloupe.1.f
			}
		}
	}

	duration = 3

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	trigger = {
	}

	immediate = {
		# Scopes set in guadeloupe.4
	}

	option = {
		name = guadeloupe.1.a
		default_option = yes
		add_modifier = {
			name = guadeloupe_fra_big_money
			months = 12
			is_decaying = no
		}
		scope:guadeloupe_buyer = {
			trigger_event = { id = guadeloupe.2 days = 0 popup = yes }
		}
		#start working from here
		show_as_tooltip = {
			random_state = {
				limit = {
					state_region = s:STATE_West_Indies
					owner = scope:guadeloupe_seller
				}
				state_region = {
					remove_claim = root
					add_claim = scope:guadeloupe_buyer
				}
				set_state_owner = scope:guadeloupe_buyer
			}
			change_relations = {
				country = scope:guadeloupe_buyer
				value = 10
			}
		}
		ai_chance = {
			base = 75
		}
		#--------------------------------------#

	}
	option = {
		name = guadeloupe.1.b
		scope:guadeloupe_buyer = {
			trigger_event = { id = guadeloupe.3 days = 0 popup = yes }
		}
		show_as_tooltip = {
			change_relations = {
				country = scope:guadeloupe_buyer
				value = -20
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					scope:guadeloupe_seller = { 
						NOT = {
							has_variable = guadeloupe_sale_cooldown #Unsolicited offer
						}
					}
				}
				add = 25
			}
		}
	}
}
