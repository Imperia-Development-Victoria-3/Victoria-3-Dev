#Opium Epidemic JEs
je_the_opium_dilemma = {
	#Sichuan
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_historical_content
	complete = {
		any_scope_building = {
			is_building_type = building_opium_plantation
			cash_reserves_ratio > 0.5
			weekly_profit > 0
			occupancy >= 0.75
			level >= 10
		}
	}
	on_complete = {
		c:CHI ?= {
			trigger_event = {
				#Notification event
				id = imperia_china_opium.1
			}
		}
		add_journal_entry = {
			type = je_chinese_opium_trade
		}
		add_modifier = {
			name = modifier_opium_profits
		}
	}
}

je_opium_trade_participation = {
	#For Substates once Sichuan initializes
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_historical_content
	complete = {
		any_scope_building = {
			is_building_type = building_opium_plantation
			cash_reserves_ratio > 0.5
			weekly_profit > 0
			occupancy >= 0.75
			level >= 10
		}
	}
	on_complete = {
		add_journal_entry = {
			type = je_chinese_opium_trade
		}
		add_modifier = {
			name = modifier_opium_profits
		}
	}
	invalid = {
		OR = {
			has_global_variable = china_shatters
			has_global_variable = opium_epidemic_resolved
		}
	}
}

je_chinese_opium_trade = {
	#JE w/modifier to denote participation in the Chinese Opium Trade
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_historical_content
	complete = {
		any_scope_building = {
			is_building_type = building_opium_plantation
			level > 5
			count = 0
		}
	}
	on_complete = {
		add_journal_entry = {
			type = je_opium_trade_participation
		}
		remove_modifier = modifier_opium_profits
	}
	fail = {
		c:CHI ?= {
			OR = {
				has_modifier = modifier_opium_ascendant
				has_modifier = modifier_opium_eradicated
			}
		}
	}
	on_fail = {
		remove_modifier = modifier_opium_profits
	}
	invalid = {
		OR = {
			has_global_variable = china_shatters
			has_global_variable = opium_epidemic_resolved
		}
	}
	on_invalid = {
		remove_modifier = modifier_opium_profits
	}
}

je_the_opium_struggle = {
	#China opium epidemic JE, part of Fragile Unity
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_historical_content
	immediate = {
		set_variable = {
			name = opium_struggle_var
			value = 0
		}
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		any_subject_or_below = {
			has_modifier = modifier_opium_profits
			count = 0
		}
		NOT = {
			has_law = law_type:law_free_trade
		}
		this.market = {
			any_trade_route = {
				goods = g:opium
				count = 0
				level > 6
			}
		}
	}
	on_complete = {
		trigger_event = {
			#Opium Eradicated
			id = imperia_china_opium.2
		}
		custom_tooltip = {
			text = fragile_unity_major_remove_tt
		}
	}
	timeout = 16425
	on_timeout = {
		trigger_event = {
			#Opium Ascendant
			id = imperia_china_opium.3
		}
		custom_tooltip = {
			text = fragile_unity_major_add_tt
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					any_subject_or_below = {
						has_modifier = modifier_opium_profits
						count = 0
					}
					NOT = {
						has_law = law_type:law_free_trade
						any_market = {
							NOT = {
								owner = root
							}
							any_trade_route = {
								exporter.owner = root
								goods = g:opium
							}
							owner = {
								any_primary_culture = {
									NOT = {
										has_cultural_obsession = opium
									}
								}
							}
						}
					}
				}
				change_variable = {
					name = opium_struggle_var
					add = 1
				}
			}
		}
	}
	current_value = {
		value = var:opium_struggle_var
	}
	goal_add_value = {
		add = 60
	}
	invalid = {
		has_global_variable = china_shatters
	}
	progressbar = yes
}
