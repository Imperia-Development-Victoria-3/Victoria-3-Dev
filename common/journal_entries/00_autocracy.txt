je_autocracy = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	group = je_group_internal_affairs
	is_shown_when_inactive = {
		is_autocratic = yes
	}
	possible = {
		has_technology_researched = nationalism
		OR = {
			ig:ig_landowners = {
				is_powerful = yes
			}
			ig:ig_devout = {
				is_powerful = yes
			}
		}
	}
	immediate = {
		set_variable = {
			name = autocracy_var
			value = 1
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					is_autocratic = yes
					NOT = {
						any_civil_war = {
							civil_war_progress > 0.5
						}
					}
				}
				change_variable = {
					name = autocracy_var
					add = 1
				}
			}
		}
		random_events = {
			10 = 0
			1 = autocracy.2			#Secret Societies event. Punish or let be an IG against authoritarianism - Intelligentsia only?
			1 = autocracy.3			#Blessed are the Meek. Extra loyalists/less radicals from SoL changes
		}
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		is_autocratic = yes
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.5
			}
		}
	}
	on_complete = {
		trigger_event = {
			id = autocracy.1
		}
		set_variable = {
			name = chose_autocracy
		}
	}
	fail = {
		is_autocratic = no
	}
	current_value = {
		value = {
			value = 0
			# Simple var check here removes thousands of errors
			if = {
				limit = {
					has_variable = autocracy_var
				}
				add = root.var:autocracy_var
			}
		}
	}
	goal_add_value = {
		add = 360
	}
	progressbar = yes
	weight = 100
}
