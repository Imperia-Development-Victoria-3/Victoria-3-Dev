imperia_quick_enact_law = {
	scope = law
	saved_scopes = {
		country
	}

	effect = {
		scope:country = {
			if = {
				limit = {
					NOT = {
						has_law = prev.type
					}
				}
				custom_tooltip = {
					text = EXECUTIVE_ORDER_DESC
					activate_law = prev.type
				}
				add_modifier = {
					name = modifier_imperia_foreign_policy_cooldown
					months = 6
				} 
			}
		}
	}

	is_valid = {
		scope:country = {
			NOT = {
				has_law = prev.type
				has_modifier = modifier_imperia_foreign_policy_cooldown
			}
			aggressive_diplomatic_plays_permitted = yes
		}

		trigger_if = {
			limit = {
				type = law_type:law_anti_imperialist
			}
			scope:country = {
				NOT = {
					OR = {
						has_law = law_type:law_colonial_exploitation
						has_law = law_type:law_colonial_resettlement	
					}
				}
			}
		}

		trigger_if = {
			limit = {
				type = law_type:law_pacifist
			}
			scope:country = {
				NOT = {
					OR = {
						has_law = law_type:law_professional_army
						has_law = law_type:law_national_guard
						has_law = law_type:law_militarized_police
					}
				}
			}
		}
	}

	is_shown = {
		type != law_type:law_true_neutrality
	}
}
