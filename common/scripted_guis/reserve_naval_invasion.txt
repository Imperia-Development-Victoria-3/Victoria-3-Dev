reserve_state = {
	saved_scopes = { 
		invaded_state
	} 
	
    effect = {
		scope:invaded_state.owner = {
			every_scope_state = {
				limit = {
					this.region = scope:invaded_state.region
					is_coastal = yes
				}
				add_modifier = {
					name = modifier_incoming_naval_invasion
					days = 50
				}
			}
		}
    }
}
