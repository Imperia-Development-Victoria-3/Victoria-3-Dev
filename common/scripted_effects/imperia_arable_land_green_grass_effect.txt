imperia_arable_land_green_grass_effect = {
	remove_modifier ?= imperia_greener_grass_modifier
	if = {
		limit = {
			has_decree = decree_greener_grass_campaign
		}
		add_modifier = {
			name = imperia_greener_grass_modifier
			multiplier = {
				value = free_arable_land
				multiply = modifier:state_migration_from_arable_land_mult
			}
		}
	}
}
