	progressiveness = -80
	modifier = {
		country_infamy_decay_mult = -0.1
		country_infamy_generation_mult = 0.1
		country_influence_add = -150
		country_radicals_from_conquest_mult = 0.3
	}

	on_activate = {
		custom_tooltip = {
			text = custom_imperia_law_expansionism_tip
		}		
	}

	unlocking_technologies = {
	}
	disallowing_laws = {
		law_anti_imperialist
		law_armed_neutrality
		law_imperialism
		law_pacifist
		law_true_neutrality
	}
	possible_political_movements = {
	}
	pop_support = {
		value = 0
	}
	ai_will_do = {
		exists = ruler
		ruler = {
			has_trait = imperious
		}
		NOT = {
			has_law = law_type:law_true_neutrality
		}
		ruler = {
			OR = {
				has_ideology = ideology:ideology_royalist
				has_ideology = ideology:ideology_jingoist_leader
				has_ideology = ideology:ideology_fascist
				has_ideology = ideology:ideology_ethno_nationalist
			}
		}
	}
}
